## Overview

The `useModulos` hook manages module data and provides formatted options suitable for PrimeReact Dropdown components. It simulates loading module data and transforms it into a format with `label` and `value` properties.

## Source Code

```typescript hooks/useModulos.ts
import { useState, useEffect } from "react";

// Define el tipo de módulo
type Modulo = {
  id: number;
  nombre: string;
};

export const useModulos = () => {
  // Datos simulados
  const modulos_select: Modulo[] = [
    { id: 0, nombre: "Oficinas centrales" },
    { id: 1, nombre: "Modulo 1" },
    { id: 2, nombre: "Modulo 2" },
    { id: 3, nombre: "Modulo 3" },
    { id: 4, nombre: "Modulo 4" },
    { id: 5, nombre: "Modulo 5" },
    { id: 6, nombre: "Modulo 6" },
    { id: 7, nombre: "Modulo 7" },
  ];

  // Tipa el estado como array de Modulo
  const [modulos, setModulos] = useState<Modulo[]>([]);

  useEffect(() => {
    // Simula carga de datos
    setModulos(modulos_select);
  }, []);

  // Opciones para un select
  const modulosOptions = modulos.map((modulo) => ({
    ...modulo,
    label: modulo.nombre,
    value: modulo.id,
  }));

  return { modulosOptions };
};
```

## Type Definitions

### Modulo Type

The base module data structure:

  <ResponseField name="nombre" type="string" required>

### ModuloOption Type

The formatted option structure for dropdowns (extends Modulo):

<ResponseField name="ModuloOption" type="object">
  <ResponseField name="id" type="number" required>
    Unique identifier for the module
  </ResponseField>
  
  <ResponseField name="nombre" type="string" required>
    Display name of the module
  </ResponseField>
  
  <ResponseField name="label" type="string" required>
    Label property for dropdown display (same as nombre)
  </ResponseField>
  
  <ResponseField name="value" type="number" required>
    Value property for dropdown selection (same as id)
  </ResponseField>
</ResponseField>

## Return Values

<ResponseField name="modulosOptions" type="ModuloOption[]">
  Array of module objects formatted for PrimeReact Dropdown components. Each option includes all original module properties plus `label` and `value` properties.
</ResponseField>

## Usage Examples

### Basic Usage

```tsx
import { useModulos } from '../hooks/useModulos';
import { Dropdown } from 'primereact/dropdown';
import { useState } from 'react';

export const ModuleSelector = () => {
  const { modulosOptions } = useModulos();
  const [selectedModule, setSelectedModule] = useState(null);
  
  return (
    <Dropdown
      value={selectedModule}
      onChange={(e) => setSelectedModule(e.value)}
      options={modulosOptions}
      placeholder="Select a module"
    />
  );
};
```

### With Type Safety

```tsx
import { useModulos } from '../hooks/useModulos';
import { Dropdown } from 'primereact/dropdown';
import { useState } from 'react';

export const TypedModuleSelector = () => {
  const { modulosOptions } = useModulos();
  const [selectedModule, setSelectedModule] = useState<number | null>(null);
  
  const handleChange = (moduleId: number) => {
    setSelectedModule(moduleId);
    console.log('Selected module ID:', moduleId);
    
    // Access full module data
    const module = modulosOptions.find(m => m.id === moduleId);
    console.log('Module name:', module?.nombre);
  };
  
  return (
    <Dropdown
      value={selectedModule}
      onChange={(e) => handleChange(e.value)}
      options={modulosOptions}
      optionLabel="label"
      placeholder="Select a module"
    />
  );
};
```

## Implementation Details

### Data Loading

The hook uses `useEffect` to simulate data loading on mount. In a production environment, this could be replaced with an API call:

```tsx
useEffect(() => {
  // Example: Replace with real API call
  fetchModulosFromAPI().then(data => setModulos(data));
}, []);
```

### Option Formatting

The hook transforms module data to match PrimeReact Dropdown requirements:

```tsx
const modulosOptions = modulos.map((modulo) => ({
  ...modulo,        // Preserves original properties (id, nombre)
  label: modulo.nombre,  // Adds label for display
  value: modulo.id,      // Adds value for selection
}));
```

This ensures compatibility with PrimeReact's Dropdown component while maintaining access to all original module properties.
